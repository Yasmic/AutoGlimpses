2 Casting in C++

2.1 const_cast<Type>(variable)

const_cast Its used to change the const or volatile qualifiers of pointers or referances . where type should be either pointer, reference or pointer to member type

int main()
{
    int a=5;
    const int *p=&a;
    int *b=const_cast<int*>(p)
    *b=10;
    cout<<*b<<endl;
    return 0;
}
  • Result=10;

2.2 static_cast<Type>(variable)

Its an implicity conversion in c++

int main()
{
    char c='a';
    int data=static_cast<int>(a);
    cout<<data<<endl;
    return 0;
}
  • Result=10;

2.3 reinterpret_cast<Type>(variable)

It is used to cast any pointer type to any pointer type

int main()
{
    int a=5;
    int *iptr=&a;
    char *cptr=reinterpret_cast<char*>(&iptr);
    cout<<*cptr<<endl;
    return 0;
}

2.4 dynamic_cast<Type>(variable)

Its used for safe downcasting at run time. The dynamic_cast to work atleast one virtual function is necessory in the base class

class base
{
    public :
    virtual void print()
    {
        cout<<"hey"<<endl;
    }
};

class derived : public base
{
    public :
    virtual void print()override
    {
        cout<<"hey derived"<<endl;
    }
};
int main()
{
    derived d_obj;
    base *b_obj=dynamic_cast<derived*>(&d_obj);
    b_obj->print();
}
  • Result=hey derived
